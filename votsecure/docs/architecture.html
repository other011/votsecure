votsecure/
├── blockchain/                    ← Smart contracts Ethereum
│   ├── contracts/
│   │   └── VotSecure.sol        ← Contract principal (Solidity 0.8)
│   ├── scripts/
│   │   └── deploy.js            ← Script deploy Hardhat
│   ├── test/
│   │   └── VotSecure.test.js    ← Teste contract (Chai/Mocha)
│   └── hardhat.config.js
│
├── backend/                       ← API Node.js/Express
│   ├── src/
│   │   ├── auth/
│   │   │   └── authService.js   ← Register, Login, JWT
│   │   ├── vote/
│   │   │   └── voteService.js   ← castVote, tally, receipt
│   │   ├── crypto/
│   │   │   └── cryptoService.js ← SHA-256, AES-GCM, HMAC
│   │   ├── validation/
│   │   │   └── cnpValidator.js  ← Validare CNP (13 reguli)
│   │   ├── blockchain/
│   │   │   └── blockchainService.js ← ethers.js interface
│   │   ├── db/
│   │   │   └── pool.js          ← PostgreSQL connection pool
│   │   ├── middleware/
│   │   │   └── authMiddleware.js ← JWT verify, role check
│   │   └── audit/
│   │       ├── logger.js        ← Winston logging
│   │       └── auditService.js  ← audit_log → PostgreSQL
│   ├── routes/
│   │   ├── auth.routes.js       ← POST /register, /login
│   │   ├── vote.routes.js       ← POST /cast, GET /results
│   │   └── admin.routes.js      ← Admin endpoints
│   ├── tests/
│   │   ├── unit/
│   │   │   ├── cnpValidator.test.js
│   │   │   └── cryptoService.test.js
│   │   └── integration/
│   │       └── auth.test.js
│   ├── server.js                    ← Entry point Express
│   └── package.json
│
├── frontend/                      ← React.js (Vite)
│   └── src/
│       ├── components/
│       │   └── auth/
│       │       └── RegisterForm.jsx ← CNP validation UI
│       └── utils/
│           ├── cnpValidator.js  ← Mirror validare backend
│           └── apiClient.js     ← Fetch wrapper + JWT
│
└── config/                        ← Configurare globală
    ├── .env.example                ← Template variabile mediu
    ├── schema.sql                   ← Schema PostgreSQL
    ├── contract-address.json        ← Generat după deploy
    └── VotSecure.abi.json           ← Generat după compile
